  <form class="input-area" id="form">
    <input type="text" id="prompt" placeholder="כתבי הודעה..." required />
    <button class="send-btn" type="submit">➤</button>
  </form>

  <script>
    const form = document.getElementById('form');
    const input = document.getElementById('prompt');
    const chat = document.getElementById('chat');

    // שינוי כיוון שפה לפי הקלט
    input.addEventListener('input', () => {
      const firstChar = input.value.trim()[0];
      input.dir = /[\u0590-\u05FF]/.test(firstChar) ? 'rtl' : 'ltr';
    });

    // שליחת הודעה
    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      const userText = input.value.trim();
      if (!userText) return;

      addBubble(userText, 'user');
      input.value = '';

      const iframe = document.createElement('iframe');
      iframe.className = 'anim-container';
      chat.appendChild(iframe);

      const response = await fetch('/api/generate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt: userText }),
      });
      const html = await response.text();
      iframe.srcdoc = html;

      setTimeout(() => {
        iframe.remove();
        const bubble = addBubble(stripHTML(html), 'assistant');
        bubble.onclick = () => replayAnimation(html);
      }, 6000); // כמה שניות לפני שמופיעה בועה רגילה
    });

    // הוספת בועה רגילה
    function addBubble(text, role) {
      const div = document.createElement('div');
      div.className = 'bubble ' + role;
      div.textContent = text;
      chat.appendChild(div);
      chat.scrollTop = chat.scrollHeight;
      return div;
    }

    // הסרת תגיות HTML
    function stripHTML(html) {
      const tmp = document.createElement('div');
      tmp.innerHTML = html;
      return tmp.textContent || tmp.innerText || '';
    }

    // הצגת אנימציה מחדש בלחיצה
    function replayAnimation(html) {
      const iframe = document.createElement('iframe');
      iframe.className = 'anim-container';
      chat.appendChild(iframe);
      iframe.srcdoc = html;
      setTimeout(() => iframe.remove(), 6000);
    }
  </script>
